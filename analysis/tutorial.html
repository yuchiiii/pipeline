iter <- c(2, 5, 10)
rf_result <- data.frame(matrix(ncol = 3, nrow = 30))

  k <- iter[i]
  for (j in 1:30) {
    ans <- my_rf_cv(k)
    rf_result[j,i] <- ans
  }
}
